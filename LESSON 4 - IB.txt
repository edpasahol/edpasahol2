function showEventInfo(eventType) {
    let infoText = "";
    if(eventType === "alert") {
        infoText = "Alert Box: Displays a message that the user must acknowledge.";
    } else if(eventType === "prompt") {
        infoText = "Prompt Box: Asks the user to enter input before proceeding.";
    } else if(eventType === "confirm") {
        infoText = "Confirm Box: Requests user confirmation to proceed or cancel.";
    } else if(eventType === "ifelse") {
        infoText = "Conditional Statements: Executes code blocks based on evaluated conditions.";
    }
    document.getElementById("eventInfo").innerHTML = infoText;
}

function demoAlert() {
    alert("This is an alert box!");
}

function demoPrompt() {
    let name = prompt("Enter your name:", "Guest");
    if(name !== null) {
        document.getElementById("userOutput").innerHTML = "Hello, " + name + "!";
    } else {
        document.getElementById("userOutput").innerHTML = "You cancelled the prompt.";
    }
}

function demoConfirm() {
    let proceed = confirm("Do you want to proceed?");
    document.getElementById("userOutput").innerHTML = proceed ? "You clicked OK!" : "You clicked Cancel!";
}

function demoConditional() {
    let coins = parseInt(prompt("Enter number of coins you have:", "0"));
    let output = "";
    if(coins === 5) {
        output = "Buy the Orange!";
    } else if(coins < 4) {
        output = "Buy the Apple!";
    } else {
        output = "Buy nothing!";
    }
    document.getElementById("userOutput").innerHTML = output;
}

function confirmBack() {
    let userConfirmed = confirm("Are you sure you want to go back to the main page?");
    if(userConfirmed) {
        window.location = "homepage.html";
    }
    // If canceled, do nothing
}
